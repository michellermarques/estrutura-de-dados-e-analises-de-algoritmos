#include <stdio.h>

// Função para calcular a média de três notas
float calcularMedia(float nota1, float nota2, float nota3) {
    return (nota1 + nota2 + nota3) / 3;
}

// Função para calcular a idade atual e a idade em 2050
void calcularIdades(int anoNascimento, int anoAtual) {
    int idadeAtual = anoAtual - anoNascimento;
    int idade2050 = 2050 - anoNascimento;
    printf("Idade atual: %d anos\n", idadeAtual);
    printf("Idade em 2050: %d anos\n", idade2050);
}

// Função para converter dólares em reais
float converterDolaresParaReais(float cotacao, float dolares) {
    return cotacao * dolares;
}

// Função para calcular o novo salário com aumento de 25%
float calcularNovoSalario(float salario) {
    return salario * 1.25;
}

// Função para calcular a área de um losango
float calcularAreaLosango(float diagonalMaior, float diagonalMenor) {
    return (diagonalMaior * diagonalMenor) / 2;
}

// Função para converter temperatura de Celsius para Fahrenheit
float converterCelsiusParaFahrenheit(float celsius) {
    return (celsius * 1.8) + 32;
}

// Função para calcular a quantidade de salários mínimos
float calcularQuantidadeSalariosMinimos(float salarioMinimo, float salario) {
    return salario / salarioMinimo;
}

// Função para calcular o novo peso após engordar ou emagrecer
float calcularNovoPeso(float peso, float percentual) {
    return (percentual > 0) ? (peso * (1 + percentual/100)) : (peso * (1 - (-percentual)/100));
}

// Função para calcular a hipotenusa de um triângulo
float calcularHipotenusa(float cateto1, float cateto2) {
    return sqrt(cateto1 * cateto1 + cateto2 * cateto2);
}

// Função para calcular o comprimento, área e volume de uma esfera
void calcularEsfera(float raio) {
    const float PI = 3.14159265359;
    float comprimento = 2 * PI * raio;
    float area = PI * raio * raio;
    float volume = (3.0 / 4.0) * PI * raio * raio * raio;

    printf("Comprimento da esfera: %.2f\n", comprimento);
    printf("Área da esfera: %.2f\n", area);
    printf("Volume da esfera: %.2f\n", volume);
}
