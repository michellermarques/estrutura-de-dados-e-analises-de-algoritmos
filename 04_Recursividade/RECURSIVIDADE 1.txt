RECURSIVIDADE 1

#include <stdio.h>

// Função recursiva para calcular a potência
int calcularPotencia(int base, int expoente) {
    if (expoente == 0) {
        return 1; // Qualquer número elevado a 0 é 1
    } else if (expoente < 0) {
        // Se o expoente for negativo, invertemos a base e tornamos o expoente positivo
        base = 1 / base;
        expoente = -expoente;
    }

    // Caso geral: base elevada ao expoente é igual a base multiplicada pela potência da base (expoente - 1)
    return base * calcularPotencia(base, expoente - 1);
}

int main() {
    int base, expoente;
    
    printf("Digite a base: ");
    scanf("%d", &base);
    
    printf("Digite o expoente: ");
    scanf("%d", &expoente);
    
    int resultado = calcularPotencia(base, expoente);
    
    printf("%d elevado a %d é igual a %d\n", base, expoente, resultado);
    
    return 0;
}
